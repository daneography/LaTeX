\section{Distributions} \\
\textbf{BERNOULLI DISTRIBUTION} \\
\indent $P(X = 1) = p, P(X = 0) = 1-p \\
\indent $pmf- $ p^x(1-p)^{1-x} \\
\indent \mu = p \\
\indent \sigma^2 = p(1-p) \\
\textbf{BINOMIAL DISTRIBUTION} \\
\indent $X \sim Bin(n,p)$ \\
\indent \mu = np$ \\
\indent \sigma^2 = np(1-p) \\
\indent P(k) = P(X = k) = {n \choose k} p^k(1-p)^{n-k} \\
\textbf{POISSON DISTRIBUTION} \\
\indent $X \sim P(\lambda) \rightarrow p(k) = e^{-\lambda}\frac{\lambda^k}{k!} \\
\indent \mu = \lambda \\
\indent \sigma^2 = \lambda \\
\indent $P$(np) \approx $ Bin$(n,p) \\
\textbf{GEOMETRIC DISTRIBUTION} \\
\indent $X \sim $ Geometric$(p) \rightarow p(k) = (1-p)^{k-1}p \\
\indent \mu = \frac{1}{p} \\
\indent \sigma^2 = \frac{1-p}{p^2} \\
\indent $\cdot$Geometric distribution is memoryless \\
\textbf{NEGATIVE BINOMIAL DISTIBUTION} \\
\indent $X \sim $ NB$(r,p) \rightarrow  p(k) = {k-1 \choose r-1}(1-p)^{k-r}p^r \\
\indent X \sim $ Geometric$(p) = X \sim $ NB$(1,p) \\
\indent \mu = \frac{r}{p} \\
\indent \sigma^2 = \frac{r(1-p)}{p^2}

F(X) = CDF, f(x) = pdf \\
For a pdf to be valid, \\

$Expected value$\\

E[X + Y] = E[X] + E[Y] \\
E[aX + b] = aE[x] + b \\
Var(aX + b) = a^2Var(x) \\
